1. What is JavaScript?

JavaScript is a programming language that adds interactivity, functionality, and dynamic behavior to websites. Unlike HTML and CSS which handle structure and styling, JavaScript allows you to create responsive elements that react to user actions like clicks, form submissions, and keyboard input.

JavaScript is the most popular programming language in the world and runs in every modern web browser. It can be used for frontend development (with frameworks like React), backend development (with Node.js), mobile apps, desktop apps, and even machine learning.

2. Variables & Data Types

📦 VARIABLES & TYPES
// Declaring Variables (USE let and const!)
let name = "Brian";        // Can be changed

4. Functions

⚙️ FUNCTIONS
// Function Declaration
function greet(name) {
    return `Hello, ${name}!`;
}
console.log(greet("Brian"));  // "Hello, Brian!"

// Arrow Function (Modern ES6)
const add = (a, b) => a + b;
console.log(add(5, 3));  // 8

// Function with default parameters
function calculateFee(amount, rate = 0.01) {
    return amount * rate;
}
console.log(calculateFee(1000));  // 10 (M-Pesa 1% fee!)

// Real Example: M-Pesa Calculator
function mpesaWithdraw(amount) {
    const fee = amount * 0.01;
    const total = amount - fee;
    return {amount, fee, total};
}
console.log(mpesaWithdraw(1000));
// {amount: 1000, fee: 10, total: 990}

5. Conditionals

// if, else if, else
let age = 18;

if (age >= 18) {
    console.log("You can vote!");
} else if (age >= 16) {
    console.log("Almost there!");
} else {
    console.log("Too young to vote.");
}

// Ternary Operator (short form)
let canVote = age >= 18 ? "Yes" : "No";

// Switch Statement
let day = "Monday";
switch(day) {
    case "Monday":
        console.log("Start of week!");
        break;
    case "Friday":
        console.log("Weekend incoming!");
        break;
    default:
        console.log("Regular day");
}

6. Loops

// For Loop
for (let i = 0; i < 5; i++) {
    console.log(`Count: ${i}`);
}

// While Loop
let count = 0;
while (count < 5) {
    console.log(count);
    count++;
}

// For...of (arrays)
const colors = ["red", "green", "blue"];
for (let color of colors) {
    console.log(color);
}

// ForEach (array method)
colors.forEach((color, index) => {
    console.log(`${index}: ${color}`);
});

9. DOM Manipulation

The DOM (Document Object Model) represents your HTML as objects that JavaScript can interact with. 

// Selecting Elements
const title = document.getElementById("title");
const buttons = document.getElementsByClassName("btn");
const firstLink = document.querySelector("a");
const allLinks = document.querySelectorAll("a");

// Changing Content
title.textContent = "New Title!";
title.innerHTML = "Bold Title";
title.style.color = "red";
title.style.fontSize = "24px";

// Creating Elements
const newDiv = document.createElement("div");
newDiv.textContent = "I am new!";
newDiv.className = "card";
document.body.appendChild(newDiv);

// Removing Elements
title.remove();

// Adding/Removing Classes
title.classList.add("highlight");
title.classList.remove("old-class");
title.classList.toggle("active");

10. Events

// Click Event
button.addEventListener("click", () => {
    alert("Button clicked!");
});

// Form Submit
form.addEventListener("submit", (e) => {
    e.preventDefault();  // Stop page refresh
    console.log("Form submitted!");
});

// Input Change
input.addEventListener("input", (e) => {
    console.log("Typing:", e.target.value);
});

// Keyboard Events
document.addEventListener("keydown", (e) => {
    if (e.key === "Enter") {
        console.log("Enter pressed!");
    }
});

// Real Example: Live Character Counter
textarea.addEventListener("input", () => {
    const count = textarea.value.length;
    counter.textContent = `${count} characters`;
});

✅ JavaScript Best Practices

✅ DO THIS
  • Use const and let (never var)
  • Use arrow functions
  • Use template literals
  • Use meaningful variable names
  • Keep functions small and focused
  • Comment complex logic
❌ AVOID THIS
  • Global variables
  • var keyword
  • Magic numbers
  • Very long functions
  • Ignoring error handling
  • eval() function
