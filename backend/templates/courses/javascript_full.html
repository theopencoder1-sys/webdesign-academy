5. Arrays & Array Methods

// Add/Remove

fruits.unshift("grape");    // Add to start
fruits.shift();             // Remove from start

// Array Methods (Functional!)
fruits.forEach(f => console.log(f));

6. Async JavaScript (Promises & Fetch)

// Fetch API
fetch('https://api.example.com/users')
    .then(res => res.json())
    .then(data => console.log(data))
    .catch(err => console.error(err));

// Async/Await (BEST!)
async function getUsers() {
    try {
        const response = await fetch('/api/users/');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('Error:', error);
    }
}

// Promise
const promise = new Promise((resolve, reject) => {
    setTimeout(() => resolve("Done!"), 2000);
});

🎯 Build: M-Pesa Calculator

<input type="number" id="amount" placeholder="Enter amount">
<button onclick="calculate()">Calculate Fee</button>
<p id="result"></p>

<script>
function calculate() {
    const amount = document.getElementById('amount').value;
    const fee = amount * 0.01;  // 1% M-Pesa fee
    const total = Number(amount) + fee;
    document.getElementById('result').textContent = 
        `Amount: Ksh ${amount}\nFee: Ksh ${fee}\nTotal: Ksh ${total}`;
}
</script>
